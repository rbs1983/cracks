const express = require("express"); const cors = require("cors"); const path = require("path"); const app = express(); app.use(cors()); app.use(express.json()); // Servir ficheiros estáticos app.use(express.static(path.join(__dirname, "public"))); app.get("/", (req, res) => { res.send("API da Liga Portuguesa está online!"); }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`Servidor a correr na porta ${PORT}`));
